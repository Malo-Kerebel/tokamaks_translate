\documentclass{standalone}

\usepackage{tikz}

\usetikzlibrary{patterns}
\usetikzlibrary{arrows.meta}

\usepackage{pgfplots}
\pgfplotsset{compat=1.18} % Assure la compatibilité avec la dernière version de pgfplots
\usepackage[utf8]{inputenc}
\usepackage[french]{babel} % Optionnel : pour la langue française
\usepackage{siunitx}
\usepackage{amsmath}
\usepackage{bm}

\begin{document}

\begin{tikzpicture}

	\draw[ -{Latex} ] (0.75, -0.25) -- (0.75, 0.25) node[midway, anchor=west] {\tiny \( \bm{E} \)};
	
	\fill (0.25, 0) circle [radius=0.025];
	\draw (0.25, 0) circle [radius=0.1] node[ anchor=north ] {\tiny \( \bm{B} \)};
	
	\pgfmathsetmacro{\length}{5};
	\pgfmathsetmacro{\angle}{15}
	\pgfmathsetmacro{\offsety}{-1}
	\pgfmathsetmacro{\xend}{\length*cos(\angle)};
	\pgfmathsetmacro{\yend}{\length*sin(\angle)+\offsety};
	\draw (0, \offsety) -- (\xend, \yend);
	
	\def\nboucle{2};
	\pgfmathsetmacro{\width}{0.5}
	\pgfmathsetmacro{\offset}{\width/2}
	\pgfmathsetmacro{\height}{2}
	
	\pgfmathsetmacro{\xinit}{1}	
	
	\foreach \i in {0, ..., \nboucle} {
		\pgfmathsetmacro{\xstart}{\xinit+\i*\width*cos(\angle)};
		\pgfmathsetmacro{\ystart}{\xinit*sin(\angle)+\offsety+\height/2*sin(\angle)*1.3+\i*\width*sin(\angle)};
		\pgfmathsetmacro{\xoffset}{\xinit+\i*\width*cos(\angle)+\offset*cos(\angle)};
		\pgfmathsetmacro{\yoffset}{\xinit*sin(\angle)+\offsety-\height/2*sin(\angle)+\i*\width*sin(\angle)+\offset*sin(\angle)};
		\pgfmathsetmacro{\xend}{\xinit+\i*\width*cos(\angle)+\width*cos(\angle)};
		\pgfmathsetmacro{\yend}{\xinit*sin(\angle)+\offsety+\height/2*sin(\angle)*1.3+\i*\width*sin(\angle)+\width*sin(\angle)};	
		\draw (\xstart, \ystart) to[ out=\angle, in=\angle ] (\xoffset, \yoffset);
		\draw (\xoffset, \yoffset) to[ out=180-\angle, in=180-\angle ] (\xend, \yend);
	}
	\pgfmathsetmacro{\i}{\nboucle+1}
	
	\pgfmathsetmacro{\xstart}{\xinit+\i*\width*cos(\angle)};
		\pgfmathsetmacro{\ystart}{\xinit*sin(\angle)+\offsety+\height/2*sin(\angle)*1.3+\i*\width*sin(\angle)};
		\pgfmathsetmacro{\xoffset}{\xinit+\i*\width*cos(\angle)+\offset*cos(\angle)};
		\pgfmathsetmacro{\yoffset}{\xinit*sin(\angle)+\offsety-\height/2*sin(\angle)+\i*\width*sin(\angle)+\offset*sin(\angle)};
		\pgfmathsetmacro{\xend}{\xinit+\i*\width*cos(\angle)+\width*cos(\angle)};
		\pgfmathsetmacro{\yend}{\xinit*sin(\angle)+\offsety+\height/2*sin(\angle)*1.3+\i*\width*sin(\angle)+\width*sin(\angle)};	
		\draw (\xstart, \ystart) to[ out=\angle, in=\angle ] (\xoffset, \yoffset);
		\draw[ -{Latex} ] (\xoffset, \yoffset) to[ out=180-\angle, in=180-\angle ] (\xend, \yend);
		
	\pgfmathsetmacro{\x}{\xoffset}
	\pgfmathsetmacro{\y}{\yoffset+\height/2*sin(\angle)}	
	
	\pgfmathsetmacro{\xend}{\length*cos(\angle)};
	\pgfmathsetmacro{\yend}{\length*sin(\angle)+\offsety};	
		
	\draw[ -{Latex} ] (\x, \y) -- (\xend, \y) node[ pos=0.75, anchor=north ] {\( \frac{E}{B} \)};
	\draw[ -{Latex} ] (\xend, \y) -- (\xend, \yend) node[ midway, anchor=west ] {\( \frac{1}{\omega_c B} \frac{\text{d} E}{\text{d}t} \)} ;
	
\end{tikzpicture}

\end{document}
